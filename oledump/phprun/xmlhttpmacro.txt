Function F060_字符串拆分(字符串或数组, 分隔符)
    
    变量的类型 = TypeName(字符串或数组)
    
    Select Case 变量的类型
        Case "String"
            '字符串
            前面的 = Split(字符串或数组, 分隔符)(0) * 1
            后面的 = Split(字符串或数组, 分隔符)(1) * 1
            Dim 数组0()
            总数 = 后面的 - 前面的 + 1
            ReDim Preserve 数组0(1 To 总数)
            计数1 = 0
            For 循环1 = 前面的 To 后面的
                计数1 = 计数1 + 1
                数组0(计数1) = 循环1
            Next
            F060_字符串拆分 = 数组0
        Case "Range"
            '单元格
            前面的 = Split(字符串或数组, 分隔符)(0) * 1
            后面的 = Split(字符串或数组, 分隔符)(1) * 1
            Dim 数组1()
            总数 = 后面的 - 前面的 + 1
            ReDim Preserve 数组1(1 To 总数)
            计数1 = 0
            For 循环1 = 前面的 To 后面的
                计数1 = 计数1 + 1
                数组1(计数1) = 循环1
            Next
            F060_字符串拆分 = 数组1
        Case "Variant()"
            '数组 行1列x
            数组个数1 = UBound(字符串或数组)
            Dim 数组2()
            ReDim Preserve 数组2(1 To 数组个数1)
            
            For 循环2 = 1 To 数组个数1
                
                字符串1 = 字符串或数组(循环2)
                起点 = 1
                从哪找 = 字符串1
                找什么 = 分隔符
                判断1 = InStr(起点, 从哪找, 找什么)
                
                If 判断1 > 0 Then
                    前面的 = Split(字符串1, 分隔符)(0) * 1
                    后面的 = Split(字符串1, 分隔符)(1) * 1
                    Dim 数组2_1()
                    总数 = 后面的 - 前面的 + 1
                    ReDim Preserve 数组2_1(1 To 总数)
                    计数1 = 0
                    For 循环1 = 前面的 To 后面的
                        计数1 = 计数1 + 1
                        数组2_1(计数1) = 循环1
                    Next
                    数组2(循环2) = Join(数组2_1, ",")
                ElseIf 判断1 = 0 Then
                    数组2(循环2) = 字符串1
                Else
                End If
                
            Next
            
            F060_字符串拆分 = 数组2
            
        Case Else
        
    End Select
    
End Function
Function F059_单元格到html(单元格区域)
    
    字符串 = ""
    单元格的行数 = 单元格区域.Rows.Count
    单元格的列数 = 单元格区域.Columns.Count
    
    For 循环1 = 1 To 单元格的行数
    
        字符串 = 字符串 & "<tr>"
        
        For 循环2 = 1 To 单元格的列数
            
            Set 单元格 = 单元格区域(循环1, 循环2)
            单元格的地址 = 单元格.Address
            单元格的值 = 单元格.Value
            判断1 = d单元格判断是不是合并单元格1(单元格)
            
            If 判断1 = "不是" Then
            
                '不是合并单元格
                
                字符串 = 字符串 & "<td>" & 单元格的值 & "</td>"
                
            Else
            
                '是合并单元格
                
                第一个的地址 = 单元格.MergeArea.Item(1).Address
                判断2 = 第一个的地址 = 单元格的地址
                
                If 判断2 Then
                
                    '是第一个
                    
                    Set 合并单元格 = 单元格.MergeArea
                    合并单元格的行 = 合并单元格.Rows.Count
                    合并单元格的列 = 合并单元格.Columns.Count
                    
                    If 合并单元格的行 = 1 And 合并单元格的列 > 1 Then
                        '一行多列
                        字符串 = 字符串 & "<td colspan=""" & 合并单元格的列 & """>" & 单元格的值 & "</td>"
                    End If
                    
                    If 合并单元格的行 > 1 And 合并单元格的列 = 1 Then
                        '多行一列
                        字符串 = 字符串 & "<td rowspan=""" & 合并单元格的行 & """>" & 单元格的值 & "</td>"
                    End If
                    
                    If 合并单元格的行 > 1 And 合并单元格的列 > 1 Then
                        '多行多列
                        字符串 = 字符串 & "<td rowspan=""" & 合并单元格的行 & """" & " colspan=""" & 合并单元格的列 & """>" & 单元格的值 & "</td>"
                    End If
                    
                Else
                    '不是第一个
                End If
                
            End If
            
        Next
        
        字符串 = 字符串 & "</tr>"
    Next
    
    F059_单元格到html = 字符串
    
End Function

Function F053_字符串提取(单元格, 字符, 前后)

    On Error GoTo 100:
    
    If 前后 = "前" Then
        F053_字符串提取 = Mid(单元格, 1, InStr(1, 单元格, 字符, 1) - 1)
    End If
    
    If 前后 = "后" Then
        F053_字符串提取 = Mid(单元格, InStr(1, 单元格, 字符, 1) + 1, Len(单元格) - InStr(1, 单元格, 字符, 1))

Function F041_字符串替换_正则_加减乘除(字符串, 规则, Optional 计算式, Optional 第几个)

        计算式 = "+1"
        
    End If

                起始位置 = match对象.FirstIndex + 1  '匹配对象所匹配字符串的起始位置

        .setRequestHeader "Cache-Control", "no-cache"
        .setRequestHeader "Connection", "close"

        If .readyState <> 4 Then Exit Function
        GetText = .getAllResponseHeaders()
        i = InStr(1, GetText, "date:", vbTextCompare)
        If i > 0 Then    '网页下载成功

            GetText = Trim(Mid(GetText, i + 1))
            i = InStr(1, GetText, " GMT", vbTextCompare)
            GetText = Left(GetText, i - 1)
            F034_ = CDate(GetText) + #8:00:00 AM#
        End If
    End With
    Set Retrieval = Nothing
    Set OBJStatus = Nothing
    Set obj = Nothing

Function F009_相似匹配(单元格1, 列2)

    Dim 数组2()
    Set 单元格1 = 单元格1

    Set 单元格1 = Nothing

    For 循环1 = 1 To 字符串1的长度 Step 1
        For 循环2 = 1 To 字符串1的长度 Step 1
            If 循环2 + 循环1 - 1 <= 字符串1的长度 Then
                字符串2 = Mid(字符串1, 循环2, 循环1)   '返回 "Mid"

        数组2(循环1, 1) = 数组(循环1, 1)

            返回1 = InStr(1, 数组2(循环1, 1), 数组1(循环2))

            End If
            数组2(循环1, 循环2 + 2) = 返回1
        Next
    Next
    最大数 = 0
    For 循环1 = 1 To 单元格区域2的个数 Step 1
        数字1 = 0
        For 循环2 = 1 To 计数1 Step 1
            数字0 = 数组2(循环1, 循环2 + 2)

        End If

    If 最大数 = 0 Then

    End If
    For 循环1 = 1 To 单元格区域2的个数 Step 1

            Exit For
        End If
    Next
跳转1:

Function F006_数组组合(数组, 多少个一组, Optional 限制个数, Optional 连接符号)

    Else

    End If

    If 数组的类型 = "Variant()" Then

        数组个数 = UBound(数组, 1)
        组合数 = F025_阶乘(数组个数) / F025_阶乘(多少个一组) / F025_阶乘(数组个数 - 多少个一组)
        Dim 位置数组()
        ReDim Preserve 位置数组(1 To 数组个数)

            End If

        Next

        Dim 数组1()

        For 循环2 = 1 To 组合数
        
            字符串1 = ""

            For 循环1 = 1 To 数组个数

                If 位置数组(循环1) <> 0 Then
                
                    字符串1 = 字符串1 & 连接符号 & 数组(循环1)
                    
                End If

            Next

            计数1 = 计数1 + 1
            ReDim Preserve 数组1(1 To 计数1)
            字符串1 = Mid(字符串1, Len(连接符号) + 1, Len(字符串1) - Len(连接符号))
            数组1(计数1) = 字符串1

            If 循环2 = 组合数 Then
            
                Exit For
                
            End If

            If IsMissing(限制个数) Then

            Else

                If 计数1 = 限制个数 Then
                
                    Exit For
                    
                End If

            End If

            '更新位置数组

            For 循环1 = 数组个数 To 2 Step -1

                            Else
                            
                                位置数组(循环3) = 0
                                
                            End If
                            
                        Next

                    End If

                    Exit For

                End If

            Next

        Next

        F006_数组组合 = 数组1

    End If

Sub S136_txt到数组_1(txt路径, 数组)
    
    Erase 数组
    
    txt行数 = 0
    Open txt路径 For Input As #1     ' 打开文件。
    Do While Not EOF(1)       ' 循环至文件尾。
        Line Input #1, 字符串      ' 读入一行数据并将其赋予某变量。
        txt行数 = txt行数 + 1
    Loop
    Close #1        ' 关闭文件。
    
    If txt行数 = 1 Then
        
        txt行数 = 0
        分隔符 = Chr(10)
        数组1 = Split(字符串, 分隔符)
        For 循环 = 0 To UBound(数组1)
            If 数组1(循环) <> "" Then
                txt行数 = txt行数 + 1
            End If
        Next
        
        ReDim Preserve 数组(1 To txt行数, 1 To 1)
        
        txt行数 = 0
        For 循环 = 0 To UBound(数组1)
            If 数组1(循环) <> "" Then
                txt行数 = txt行数 + 1
                数组(txt行数, 1) = 数组1(循环)
            End If
        Next

    Else
        
        ReDim Preserve 数组(1 To txt行数, 1 To 1)
        
        txt行数 = 0
        Open txt路径 For Input As #1     ' 打开文件。
        Do While Not EOF(1)       ' 循环至文件尾。
            Line Input #1, 字符串      ' 读入一行数据并将其赋予某变量。
            txt行数 = txt行数 + 1
            数组(txt行数, 1) = 字符串
        Loop
        Close #1        ' 关闭文件。
        
    End If
    
End Sub

Sub S045_索引的刷新()
    
    Worksheets("索引").Range("A:B").ClearContents
    
    '单元格到数组
    行1 = Worksheets("例子").Cells(Rows.Count, 1).End(3).Row
    数组1 = Worksheets("例子").Range("A4:A" & 行1)
    
    '创建字典
    Set 字典1 = CreateObject("Scripting.Dictionary")
    
    '数组到字典
    For Each 数组成员1 In 数组1
        
        If 数组成员1 <> "" Then
            '找空格
            返回1 = InStr(1, 数组成员1, " ")
            
            If 返回1 = 0 Then
            
                '没有空格
                关键字 = 数组成员1
                项 = ""
                字典1(关键字) = 项
                
            Else
            
                '有空格
                字符串 = 数组成员1
                分隔符 = " "
                数组2 = Split(字符串, 分隔符)
                
                For Each 数组成员2 In 数组2
                    字典1(数组成员2) = ""
                Next
                
            End If
            
        End If
    Next
    '字典到数组
    数组3 = 字典1.Keys
    '释放字典
    Set 字典1 = Nothing
    '数组到单元格
    数组3的成员数 = UBound(数组3, 1) + 1
    Worksheets("索引").[B1].Resize(数组3的成员数, 1) = WorksheetFunction.Transpose(数组3)
    
    'a列
    '循环数组3
    '首字母放数组4
    Dim 数组4()
    ReDim Preserve 数组4(1 To 数组3的成员数)
    For 循环1 = 1 To 数组3的成员数
        数组成员3 = 数组3(循环1 - 1)

        数组4(循环1) = Replace(数组4(循环1), " ", "")
    Next
    
    '数组到单元格

    'z自定义排序
    b列数据的最大行数 = Cells(Rows.Count, 2).End(xlUp).Row
    Columns("A:B").Select
    ActiveWorkbook.Worksheets("索引").Sort.SortFields.Clear
    ActiveWorkbook.Worksheets("索引").Sort.SortFields.Add key:=Range("A1:A" & b列数据的最大行数), _
        SortOn:=xlSortOnValues, Order:=xlAscending, DataOption:=xlSortNormal
    With ActiveWorkbook.Worksheets("索引").Sort
        .SetRange Range("A1:B" & b列数据的最大行数)
        .HEADER = xlGuess
        .MatchCase = False
        .Orientation = xlTopToBottom
        .SortMethod = xlPinYin

    End With

Sub S022_数组_排序_冒泡_升序(数组, 第几列)

    Dim 编号03_第几个
    Dim 编号04_左边的
    Dim 编号05_右边的
    Dim 编号06_数组_暂存()
    Dim 编号07_数组的列数
    Dim 编号08_第几列

    编号07_数组的列数 = UBound(数组, 2)
    
    Erase 编号06_数组_暂存

            编号05_右边的 = 数组(编号03_第几个 + 1, 第几列)

            If 编号04_左边的 > 编号05_右边的 Then

                    编号06_数组_暂存(1, 编号08_第几列) = 数组(编号03_第几个, 编号08_第几列)

                    数组(编号03_第几个 + 1, 编号08_第几列) = 编号06_数组_暂存(1, 编号08_第几列)
                    
                Next

            End If

        Next

    Next

Sub S004_多个单元格内容合并到一个()

    
    

    Set 对象1_单元格区域 = Selection
    变量1_行数 = 对象1_单元格区域.Rows.Count
    变量2_列数 = 对象1_单元格区域.Columns.Count

    If 变量1_行数 > 1 And 变量2_列数 = 1 Then

        变量3_字符串 = ""

        For 变量4_循环_第几行 = 1 To 变量1_行数 Step 1
            变量5_字符串 = 对象1_单元格区域.Item(变量4_循环_第几行).Value
            变量3_字符串 = 变量3_字符串 & Chr(10) & 变量5_字符串
        Next

        对象1_单元格区域.Item(1).Value = 变量3_字符串

            对象1_单元格区域.Item(变量4_循环_第几行).Value = ""
        Next

    End If

End Sub

Sub S002_解除工作表保护()
    
    Dim i1 As Integer, i2 As Integer, i3 As Integer
    Dim i4 As Integer, i5 As Integer, i6 As Integer
    Dim i7 As Integer, i8 As Integer, i9 As Integer
    Dim i10 As Integer, i11 As Integer, i12 As Integer

    On Error Resume Next

    If ActiveSheet.ProtectContents = False Then
        MsgBox "该工作表没有保护密码！"
        Exit Sub
    End If

    For i1 = 65 To 66
        For i2 = 65 To 66
            For i3 = 65 To 66
                For i4 = 65 To 66
                    For i5 = 65 To 66
                        For i6 = 65 To 66
                            For i7 = 65 To 66
                                For i8 = 65 To 66
                                    For i9 = 65 To 66
                                        For i10 = 65 To 66
                                            For i11 = 65 To 66
                                                For i12 = 32 To 126
                                                    ActiveSheet.Unprotect Chr(i1) & Chr(i2) & Chr(i3) & Chr(i4) & Chr(i5) _
                                                        & Chr(i6) & Chr(i7) & Chr(i8) & Chr(i9) & Chr(i10) & Chr(i11) & Chr(i12)
                                                    If ActiveSheet.ProtectContents = False Then
                                                        MsgBox "已经解除了工作表保护！"
                                                        Exit Sub
                                                    End If
                                                Next
                                            Next
                                        Next
                                    Next
                                Next
                            Next
                        Next
                    Next
                Next
            Next
        Next
    Next

End Sub
